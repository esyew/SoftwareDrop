43   std::set<unsigned long long> BusinessLogic::GenerateStdNums(int cnt)
44   {
45   
46   
47       std::random_device rd;
48       std::mt19937_64 generator(rd());
49       std::uniform_int_distribution<unsigned long long> dist{1, 69};
50       std::set<unsigned long long> results;
51   
52       while(results.size() != cnt)
53       {
54         results.insert(dist(generator));
55       }
56     return results;
57   }
