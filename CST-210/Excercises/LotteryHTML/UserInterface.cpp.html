5    #include "UserInterface.h"
6    #include <iostream>
7    bool UserInterface::AskForMore()
8    {
9      bool isAnswered = false;
10     while (!isAnswered) {
11       cout << "Would you like to see more numbers? ('Y' or 'N')";
12       char ans;
13       std::cin >> ans; // Ask for user answer
14       if ((ans == 'Y') || (ans == 'y')) { // Check if lower/uppercase Yes
15         isAnswered = true;
16         return true; //Restart script at main()
17       } else if ((ans == 'N') || (ans == 'n')) {
18         isAnswered = true;
19         return false; //Pass to main() that will end script
20       } else //User did not type "y" or "n"
21         std::cout << "Error: Please type 'Y' or 'N'\n"; //Confront user
22       isAnswered = false;
23     }
24     //Default, should never run
25     return false;
26   }
