48   template <typename T>
49   using identity_t = T;

51   template <typename Pattern>
52   struct ThisRefAdjuster {
53     template <typename T>
54     using AdjustT = typename std::conditional<
55         std::is_const<typename std::remove_reference<Pattern>::type>::value,
56         typename std::conditional<std::is_lvalue_reference<Pattern>::value,
57                                   const T&, const T&&>::type,
58         typename std::conditional<std::is_lvalue_reference<Pattern>::value, T&,
59                                   T&&>::type>::type;
60   
61     template <typename MockType>
62     static AdjustT<MockType> Adjust(const MockType& mock) {
63       return static_cast<AdjustT<MockType>>(const_cast<MockType&>(mock));
64     }
65   };

111  #define GMOCK_INTERNAL_WRONG_ARITY(...)                                      \ 
112    static_assert(                                                             \ 
113        false,                                                                 \ 
114        "MOCK_METHOD must be called with 3 or 4 arguments. _Ret, "             \ 
115        "_MethodName, _Args and optionally _Spec. _Args and _Spec must be "    \ 
116        "enclosed in parentheses. If _Ret is a type with unprotected commas, " \ 
117        "it must also be enclosed in parentheses.")

119  #define GMOCK_INTERNAL_ASSERT_PARENTHESIS(_Tuple) \ 
120    static_assert(                                  \ 
121        GMOCK_PP_IS_ENCLOSED_PARENS(_Tuple),        \ 
122        GMOCK_PP_STRINGIZE(_Tuple) " should be enclosed in parentheses.")
123  
124  #define GMOCK_INTERNAL_ASSERT_VALID_SIGNATURE(_N, ...)                 \ 
125    static_assert(                                                       \ 
126        std::is_function<__VA_ARGS__>::value,                            \ 
127        "Signature must be a function type, maybe return type contains " \ 
128        "unprotected comma.");                                           \ 
129    static_assert(                                                       \ 
130        ::testing::tuple_size<typename ::testing::internal::Function<    \ 
131                __VA_ARGS__>::ArgumentTuple>::value == _N,               \ 
132        "This method does not take " GMOCK_PP_STRINGIZE(                 \ 
133            _N) " arguments. Parenthesize all types with unprotected commas.")
134  
135  #define GMOCK_INTERNAL_ASSERT_VALID_SPEC(_Spec) \ 
136    GMOCK_PP_FOR_EACH(GMOCK_INTERNAL_ASSERT_VALID_SPEC_ELEMENT, ~, _Spec)
137  
138  #define GMOCK_INTERNAL_MOCK_METHOD_IMPL(_N, _MethodName, _Constness,           \ 
139                                          _Override, _Final, _NoexceptSpec,      \ 
140                                          _CallType, _RefSpec, _Signature)       \ 
141    typename ::testing::internal::Function<GMOCK_PP_REMOVE_PARENS(               \ 
142        _Signature)>::Result                                                     \ 
143    GMOCK_INTERNAL_EXPAND(_CallType)                                             \ 
144        _MethodName(GMOCK_PP_REPEAT(GMOCK_INTERNAL_PARAMETER, _Signature, _N))   \ 
145            GMOCK_PP_IF(_Constness, const, ) _RefSpec _NoexceptSpec              \ 
146            GMOCK_PP_IF(_Override, override, ) GMOCK_PP_IF(_Final, final, ) {    \ 
147      GMOCK_MOCKER_(_N, _Constness, _MethodName)                                 \ 
148          .SetOwnerAndName(this, #_MethodName);                                  \ 
149      return GMOCK_MOCKER_(_N, _Constness, _MethodName)                          \ 
150          .Invoke(GMOCK_PP_REPEAT(GMOCK_INTERNAL_FORWARD_ARG, _Signature, _N));  \ 
151    }                                                                            \ 
152    ::testing::MockSpec<GMOCK_PP_REMOVE_PARENS(_Signature)> gmock_##_MethodName( \ 
153        GMOCK_PP_REPEAT(GMOCK_INTERNAL_MATCHER_PARAMETER, _Signature, _N))       \ 
154        GMOCK_PP_IF(_Constness, const, ) _RefSpec {                              \ 
155      GMOCK_MOCKER_(_N, _Constness, _MethodName).RegisterOwner(this);            \ 
156      return GMOCK_MOCKER_(_N, _Constness, _MethodName)                          \ 
157          .With(GMOCK_PP_REPEAT(GMOCK_INTERNAL_MATCHER_ARGUMENT, , _N));         \ 
158    }                                                                            \ 
159    ::testing::MockSpec<GMOCK_PP_REMOVE_PARENS(_Signature)> gmock_##_MethodName( \ 
160        const ::testing::internal::WithoutMatchers&,                             \ 
161        GMOCK_PP_IF(_Constness, const, )::testing::internal::Function<           \ 
162            GMOCK_PP_REMOVE_PARENS(_Signature)>*) const _RefSpec _NoexceptSpec { \ 
163      return ::testing::internal::ThisRefAdjuster<GMOCK_PP_IF(                   \ 
164          _Constness, const, ) int _RefSpec>::Adjust(*this)                      \ 
165          .gmock_##_MethodName(GMOCK_PP_REPEAT(                                  \ 
166              GMOCK_INTERNAL_A_MATCHER_ARGUMENT, _Signature, _N));               \ 
167    }                                                                            \ 
168    mutable ::testing::FunctionMocker<GMOCK_PP_REMOVE_PARENS(_Signature)>        \ 
169        GMOCK_MOCKER_(_N, _Constness, _MethodName)

262  // Note: The use of `identity_t` here allows _Ret to represent return types that
263  // would normally need to be specified in a different way. For example, a method
264  // returning a function pointer must be written as
265  //
266  // fn_ptr_return_t (*method(method_args_t...))(fn_ptr_args_t...)
267  //
268  // But we only support placing the return type at the beginning. To handle this,
269  // we wrap all calls in identity_t, so that a declaration will be expanded to
270  //
271  // identity_t<fn_ptr_return_t (*)(fn_ptr_args_t...)> method(method_args_t...)
272  //
273  // This allows us to work around the syntactic oddities of function/method
274  // types.
275  #define GMOCK_INTERNAL_SIGNATURE(_Ret, _Args)                                 \ 
276    ::testing::internal::identity_t<GMOCK_PP_IF(GMOCK_PP_IS_BEGIN_PARENS(_Ret), \ 
277                                                GMOCK_PP_REMOVE_PARENS,         \ 
278                                                GMOCK_PP_IDENTITY)(_Ret)>(      \ 
279        GMOCK_PP_FOR_EACH(GMOCK_INTERNAL_GET_TYPE, _, _Args))
280  
281  #define GMOCK_INTERNAL_GET_TYPE(_i, _, _elem)                          \ 
282    GMOCK_PP_COMMA_IF(_i)                                                \ 
283    GMOCK_PP_IF(GMOCK_PP_IS_BEGIN_PARENS(_elem), GMOCK_PP_REMOVE_PARENS, \ 
284                GMOCK_PP_IDENTITY)                                       \ 
285    (_elem)
